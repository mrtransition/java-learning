🏗️ 类与对象

定义类与创建对象

public class Car {
    // 字段
    private String brand;
    private int year;

    // 构造方法
    public Car(String brand, int year) {
        this.brand = brand;
        this.year = year;
    }

    // getter / setter
    public String getBrand() { return brand; }
    public void setBrand(String brand) { this.brand = brand; }

    // 行为方法
    public void honk() {
        System.out.println(brand + " 鸣笛！");
    }
}

// 使用
Car myCar = new Car("Tesla", 2024);
myCar.honk();              // Tesla 鸣笛！
myCar.setBrand("BMW");
System.out.println(myCar.getBrand());  // BMW

封装

使用 private 隐藏实现细节，通过 public 方法暴露接口：

public class BankAccount {
    private double balance;   // 外部不可直接访问

    public BankAccount(double initial) {
        if (initial >= 0)
            this.balance = initial;
    }

    public double getBalance() { return balance; }

    public void deposit(double amount) {
        if (amount > 0) balance += amount;
    }

    public boolean withdraw(double amount) {
        if (amount > 0 && amount <= balance) {
            balance -= amount;
            return true;
        }
        return false;
    }
}

继承

public class Animal {
    protected String name;

    public Animal(String name) { this.name = name; }

    public void speak() {
        System.out.println(name + " 发出声音");
    }
}

public class Dog extends Animal {
    public Dog(String name) {
        super(name);     // 调用父类构造方法
    }

    @Override
    public void speak() {
        System.out.println(name + " 汪汪！");
    }
}

// 使用
Animal a = new Dog("Buddy");
a.speak();                     // Buddy 汪汪！

接口

接口定义契约，实现类必须实现所有抽象方法：

public interface Flyable {
    void fly();  // 默认 public abstract
}

public class Bird implements Flyable {
    @Override
    public void fly() {
        System.out.println("鸟儿飞翔");
    }
}

// 接口作为类型
Flyable f = new Bird();
f.fly();

抽象类

public abstract class Shape {
    public abstract double area();   // 子类必须实现

    public void printArea() {         // 普通方法
        System.out.println("面积: " + area());
    }
}

public class Circle extends Shape {
    private double r;

    public Circle(double r) { this.r = r; }

    @Override
    public double area() { return Math.PI * r * r; }
}
