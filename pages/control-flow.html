🔀 流程控制

if / else if / else

int score = 85;

if (score >= 90) {
    System.out.println("优秀");
} else if (score >= 75) {
    System.out.println("良好");     // 输出
} else if (score >= 60) {
    System.out.println("及格");
} else {
    System.out.println("不及格");
}

switch 语句

int day = 3;
String dayName;

switch (day) {
    case 1: dayName = "周一"; break;
    case 2: dayName = "周二"; break;
    case 3: dayName = "周三"; break;
    default: dayName = "未知";
}
System.out.println(dayName);  // 周三

switch 表达式 Java 14+

增强 switch 支持箭头语法，无需 break，可直接赋值：

String day = "MONDAY";

int result = switch (day) {
    case "MONDAY", "FRIDAY" -> 1;
    case "WEDNESDAY" -> 2;
    default -> {
        System.out.println("其他");
        yield 0;      // 多行分支用 yield 返回值
    }
};
System.out.println(result);  // 1

for 循环

// 传统 for
for (int i = 0; i < 5; i++) {
    System.out.print(i + " ");   // 0 1 2 3 4
}

// 增强 for (for-each)
int[] numbers = {10, 20, 30};
for (int n : numbers) {
    System.out.println(n);
}

while / do-while

// while — 先判断再执行
int i = 0;
while (i < 3) {
    System.out.println(i);
    i++;
}

// do-while — 至少执行一次
int j = 0;
do {
    System.out.println("至少一次");
} while (j > 0);

break 与 continue

for (int i = 0; i < 10; i++) {
    if (i == 3) continue;   // 跳过 3，继续循环
    if (i == 7) break;      // 到 7 时退出循环
    System.out.print(i + " ");  // 0 1 2 4 5 6
}
